class VirtualPerson:
    def __init__(self, name, prompt):
        self.name = name
        self.chatbot = ChatGPT(prompt)
        self.history = []

    def speak(self, message):
        self.history.append(message)
        response = self.chatbot.generate_text(message)
        print(f"{self.name}: {response}")

    def remember(self, message):
        self.history.append(message)
